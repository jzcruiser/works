代码演示

基础用法

                    
                      //页面加载开始
                      componentDidMount() {
                        let emitter = this.props.pageUtils && this.props.pageUtils.pageEmitter;
                        let pageDisappear = this.props.pageUtils.emitterChannel.PAGE_DISAPPEAR;
                        let pageAppear = this.props.pageUtils.emitterChannel.PAGE_APPEAR;
                        if (emitter) {
                            emitter.on(pageDisappear, this.videoDisappear.bind(null,"video"));
                            emitter.on(pageAppear, this.videoAppear.bind(null,"video"));
                        }
                      }
                      //页面加载完成
                      componentWillUnmount() {
                        let emitter = this.props.pageUtils && this.props.pageUtils.pageEmitter;
                        let pageDisappear = this.props.pageUtils.emitterChannel.PAGE_DISAPPEAR;
                        let pageAppear = this.props.pageUtils.emitterChannel.PAGE_APPEAR;
                        if (emitter) {
                            emitter.off(pageDisappear, this.videoDisappear.bind(null,"video"));
                            emitter.off(pageAppear, this.videoAppear.bind(null,"video"));
                        }
                      }
